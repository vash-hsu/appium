            intB = ran.nextInt(9)+1;

        intA = intB * intA;
        int[] dataSet = myGame.mutate("/", intA, intB, 3, flag);

    }

    }
}

                                 self.desired_caps['appActivity'])

    def test_gaming_divide_10_run(self):
      self._enter_game_of("divide")
      for i in xrange(10):
